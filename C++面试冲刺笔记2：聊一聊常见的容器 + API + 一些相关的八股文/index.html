void get_splity_text(std::vector<std::string>& result) {
    std::string tr = TEXT;

    // remove punctual
    tr.erase(std::remove_if(tr.begin(), tr.end(), [](const char ch) -> bool {
                 return ::ispunct(ch);

             tr.end());
